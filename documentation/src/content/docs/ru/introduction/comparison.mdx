# Сравнение effector с другими стейт менеджерами

В этом примере мы возьмем в пример простую реализацию Counter.<br/>
В качестве UI фреймворка мы возьмем React.

:::info{title="Effector не зависит от фреймкорка !"}
При этом стоит сказать, что effector не связан с React, вы можете использовать его с любым удобным для вас фреймворком, будь то [React](https://react.dev/), [Vue](https://vuejs.org/), [Solid](https://www.solidjs.com/) и др.
:::

```jsx
import { useUnit } from "effector-react";
import { createEvent, createStore } from "effector";

const $counter = createStore(0);
const $derivedCount = $counter.map((countValue) => countValue * 2);

const incremented = createEvent();
const decremented = createEvent();

$counter.on(incremented, (state) => state + 1);
$counter.on(decremented, (state) => state - 1);

export const Counter = () => {
  const [counter, derivedCount, onIncremented, onDecremented] = useUnit([
    $counter,
    $derivedCount,
    incremented,
    decremented,
  ]);

  return (
    <div>
      <h1>Count: {counter}</h1>
      <h1>Derived count: (x2) {derivedCount}</h1>
    </div>
  );
};

function CounterWrapper(){
  const [onIncremented, onDecremented] = useUnit([
    incremented,
    decremented,
  ]);

  return (
    <div>
      <Counter>
      <button onClick={onIncremented}>Increment</button>;
      <button onClick={onDecremented}>Decrement</button>;
    </div>
  )
}
```

```jsx
import create from "zustand";

const useStore = create((set) => ({
  counter: 0,
  derivedCount: 0,
  incremented: () =>
    set((state) => {
      const newCount = state.counter + 1;
      return { counter: newCount, derivedCount: newCount * 2 };
    }),
  decremented: () =>
    set((state) => {
      const newCount = state.counter - 1;
      return { counter: newCount, derivedCount: newCount * 2 };
    }),
}));

export const Counter = () => {
  const { counter, derivedCount } = useStore((state) => ({
    counter: state.counter,
    derivedCount: state.derivedCount,
  }));

  return (
    <div>
      <h1>Count: {counter}</h1>
      <h1>Derived count: (x2) {derivedCount}</h1>
    </div>
  );
};

export const CounterWrapper = () => {
  const { incremented, decremented } = useStore((state) => ({
    incremented: state.incremented,
    decremented: state.decremented,
  }));

  return (
    <div>
      <Counter />
      <button onClick={incremented}>Increment</button>
      <button onClick={decremented}>Decrement</button>
    </div>
  );
};
```

```jsx
import { createSlice, configureStore } from "@reduxjs/toolkit";
import { useDispatch, useSelector } from "react-redux";

const counterSlice = createSlice({
  name: "counter",
  initialState: { counter: 0, derivedCount: 0 },
  reducers: {
    incremented: (state) => {
      state.counter += 1;
      state.derivedCount = state.counter * 2;
    },
    decremented: (state) => {
      state.counter -= 1;
      state.derivedCount = state.counter * 2;
    },
  },
});

const store = configureStore({
  reducer: {
    counter: counterSlice.reducer,
  },
});

export const Counter = () => {
  const { counter, derivedCount } = useSelector((state) => state.counter);
  return (
    <div>
      <h1>Count: {counter}</h1>
      <h1>Derived count: (x2) {derivedCount}</h1>
    </div>
  );
};

export const CounterWrapper = () => {
  const dispatch = useDispatch();
  return (
    <div>
      <Counter />
      <button onClick={() => dispatch(counterSlice.actions.incremented())}>Increment</button>
      <button onClick={() => dispatch(counterSlice.actions.decremented())}>Decrement</button>
    </div>
  );
};
```

```jsx
import { atom, useAtom } from "jotai";

const counterAtom = atom(0);
const derivedCountAtom = atom((get) => get(counterAtom) * 2);

export const Counter = () => {
  const [counter] = useAtom(counterAtom);
  const [derivedCount] = useAtom(derivedCountAtom);

  return (
    <div>
      <h1>Count: {counter}</h1>
      <h1>Derived count: (x2) {derivedCount}</h1>
    </div>
  );
};

export const CounterWrapper = () => {
  const [, setCounter] = useAtom(counterAtom);

  const onIncremented = () => {
    setCounter((count) => count + 1);
  };

  const onDecremented = () => {
    setCounter((count) => count - 1);
  };

  return (
    <div>
      <Counter />
      <button onClick={onIncremented}>Increment</button>
      <button onClick={onDecremented}>Decrement</button>
    </div>
  );
};
```
